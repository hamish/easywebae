 14 		CKEDITOR.fire( 'loaded' );
 15
 16 		// Process all instances created by the "basic" implementation.
 17 		var pending = CKEDITOR._.pending;
 18 		if ( pending )
 19 		{
 20 			delete CKEDITOR._.pending;
 21
 22 			for ( var i = 0 ; i < pending.length ; i++ )
 23 				CKEDITOR.add( pending[ i ] );
 24 		}
 25 	});
